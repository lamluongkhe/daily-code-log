# Source: microservice/templates/ms-a-deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: ms-a
  namespace: llkhe
spec:
  replicas: 1
  selector:
    matchLabels:
      app: ms-a
  template:
    metadata:
      labels:
        app: ms-a
    spec:
      containers:
      - name: ms-a
        image: ms-a:latest
        imagePullPolicy: Never
        ports:
        - containerPort: 8181
        env:
        - name: KAFKA_BOOTSTRAP_SERVERS
          value: "kafka-1:29092"
        - name: REDIS_HOST
          value: "redis"
        - name: REDIS_PORT
          value: "6379"
        readinessProbe:
          tcpSocket:
            port: 8181
          initialDelaySeconds: 40
        livenessProbe:
          tcpSocket:
            port: 8181
          initialDelaySeconds: 60
---
